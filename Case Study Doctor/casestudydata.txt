CREATE DATABASE OnlineDoctorConsultation;

USE OnlineDoctorConsultation;

CREATE TABLE DOCTOR (
	D_USERNAME VARCHAR(50),
    D_PASSWORD VARCHAR(25),
    DOC_ID INT PRIMARY KEY AUTO_INCREMENT,
    DOC_NAME VARCHAR(30),
    DOC_SCHEDULE DATE,
    DOC_SPECIALIZATION VARCHAR(35),
    DOC_LOCATION VARCHAR(20),
    DOC_CONTACT BIGINT
);

CREATE TABLE PATIENT (
	P_USERNAME VARCHAR(50),
    P_PASSWORD VARCHAR(25),
	PATIENT_ID INT PRIMARY KEY AUTO_INCREMENT,
    P_NAME VARCHAR(20),
    P_CONTACT BIGINT,
    P_LOCATION VARCHAR(20),
    P_CONSULTATION_SCHEDULED DATE,
    P_CONSULTATION_HISTORY TEXT,
    P_HEALTH_HISTORY TEXT
);

CREATE TABLE PATIENT_LOGIN(
	PATIENT_ID INT,
    P_USERNAME VARCHAR(50)PRIMARY KEY,
    P_PASSWORD VARCHAR(25),
    FOREIGN KEY(PATIENT_ID)REFERENCES PATIENT(PATIENT_ID)
);

CREATE TABLE DOCTOR_LOGIN(
	DOC_ID INT,
    D_USERNAME VARCHAR(50)PRIMARY KEY,
    D_PASSWORD VARCHAR(25),
    FOREIGN KEY(DOC_ID)REFERENCES DOCTOR(DOC_ID)
);

CREATE TABLE REVIEW(
R_ID INT PRIMARY KEY AUTO_INCREMENT,
CONSULTATION_EXPERIENCE TEXT,
APP_EXPERIENCE INT,
DOC_ID INT,
FOREIGN KEY(DOC_ID)REFERENCES DOCTOR(DOC_ID)
);

CREATE TABLE OBSERVATION(
OBS_ID INT PRIMARY KEY AUTO_INCREMENT,
PRESCRIPTION TEXT,
DOIGNISTIC TEXT,
MEDS_PRESCRIBED TEXT,
DOC_ID INT,
PATIENT_ID INT,
FOREIGN KEY(DOC_ID)REFERENCES DOCTOR(DOC_ID),
FOREIGN KEY(PATIENT_ID)REFERENCES PATIENT(PATIENT_ID)
);

CREATE TABLE SCHEDULES(
S_ID INT PRIMARY KEY AUTO_INCREMENT,
DOC_SCHEDULE DATE,
DOC_ID INT,
PATIENT_ID INT,
);
